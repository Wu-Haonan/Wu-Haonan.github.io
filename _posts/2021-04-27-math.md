def Hanota_step (n,s,t,r):
global i
if n == 1:
   print("Step-%03d: #%d %c -> %c"%(i,n,s,t))
   i = i + 1
else:
   Hanota_step(n-1,s,r,t)
   print("Step-%03d: #%d %c -> %c"%(i,n,s,t))
   i = i + 1
   Hanota_step(n-1,r,t,s)
